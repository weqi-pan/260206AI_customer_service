# 阶段一：MySQL 数据库集成

## 一、需求分析

### 1.1 当前项目问题

1. **无用户管理体系**：用户 ID 是随机生成的，无法关联真实用户信息
2. **设备信息静态化**：设备参数、价格等信息硬编码或分散在文档中
3. **库存无法查询**：无法查询各仓库的设备库存情况
4. **使用记录存储低效**：当前使用 CSV 文件存储，查询效率低

### 1.2 参考项目实现

**电商项目（order_agent.py）：**
```python
def query_order_with_product(order_id: str):
    """查询订单信息（含商品ID）"""
    conn = sqlite3.connect(db_path)
    cursor = conn.cursor()
    cursor.execute("SELECT status, logistics_info, ... FROM orders WHERE order_id = ?", (order_id,))
    result = cursor.fetchone()
    conn.close()
    ...
```

**核心原理：**
- 使用结构化查询语言（SQL）高效检索数据
- 支持复杂查询（JOIN、聚合、条件筛选）
- 事务支持，保证数据一致性

### 步骤2：创建数据库连接管理

**新建文件：`db/connection.py`**
```python
"""
数据库连接管理模块
遵循项目风格：单例模式 + 配置驱动 + 日志记录
"""

import mysql.connector
from mysql.connector import pooling
from typing import Optional
from utils.config_handler import load_agent_config
from utils.logger_handler import logger
import yaml
from utils.path_tool import get_abs_path


class DatabaseConnectionFactory:
    """数据库连接工厂类 - 遵循项目工厂模式"""

    _instance: Optional['DatabaseConnection'] = None

    @staticmethod
    def get_connection() -> 'mysql.connector.MySQLConnection':
        """获取数据库连接（单例模式）"""
        if DatabaseConnectionFactory._instance is None:
            DatabaseConnectionFactory._instance = DatabaseConnection()
        return DatabaseConnectionFactory._instance.get_connection()


class DatabaseConnection:
    """
    数据库连接管理类
    功能：
    1. 管理数据库连接池
    2. 提供获取连接的接口
    3. 处理连接异常
    """

    def __init__(self):
        self.config = self._load_config()
        self.connection_pool: Optional[pooling.MySQLConnectionPool] = None
        self._init_pool()
        logger.info(f"[DatabaseConnection]数据库连接池初始化成功：{self.config['database']}")

    def _load_config(self) -> dict:
        """加载数据库配置"""
        config_path = get_abs_path("config/database.yml")
        try:
            with open(config_path, "r", encoding="utf-8") as f:
                config = yaml.load(f, Loader=yaml.FullLoader)
                return config["mysql"]
        except Exception as e:
            logger.error(f"[DatabaseConnection]加载配置失败：{str(e)}")
            raise e

    def _init_pool(self):
        """初始化连接池"""
        try:
            self.connection_pool = pooling.MySQLConnectionPool(
                pool_name="sweeper_pool",
                pool_size=5,
                **self.config
            )
        except Exception as e:
            logger.error(f"[DatabaseConnection]创建连接池失败：{str(e)}")
            raise e

    def get_connection(self) -> 'mysql.connector.MySQLConnection':
        """从连接池获取一个连接"""
        try:
            return self.connection_pool.get_connection()
        except Exception as e:
            logger.error(f"[DatabaseConnection]获取连接失败：{str(e)}")
            raise e

    def test_connection(self) -> bool:
        """测试数据库连接"""
        try:
            conn = self.get_connection()
            cursor = conn.cursor()
            cursor.execute("SELECT 1")
            result = cursor.fetchone()
            cursor.close()
            conn.close()
            logger.info("[DatabaseConnection]数据库连接测试成功")
            return result[0] == 1
        except Exception as e:
            logger.error(f"[DatabaseConnection]数据库连接测试失败：{str(e)}")
            return False


# 全局便捷函数
def get_db_connection():
    """获取数据库连接的便捷函数"""
    return DatabaseConnectionFactory.get_connection()
```

```python
    """设备数据模型"""
    model_id: str
    model_name: str
    category: str
    specs: str  # JSON字符串
    description: str
    price: float
    features: str  # JSON字符串

        """根据型号ID查询设备"""
        try:
            conn = get_db_connection()
            cursor = conn.cursor(dictionary=True)
            cursor.execute(
                "SELECT model_id, model_name, category, specs, description, price, features "
                "FROM devices WHERE model_id = %s",
                (model_id,)
            )
            result = cursor.fetchone()
            cursor.close()
            conn.close()

            if result:
                return cls(**result)
            return None
        except Exception as e:
            logger.error(f"[Device.get_by_model_id]查询失败：{str(e)}")
            return None

    @classmethod
    def search_by_keyword(cls, keyword: str, limit: int = 3) -> List['Device']:
        """根据关键词搜索设备（模糊匹配）"""
        try:
            conn = get_db_connection()
            cursor = conn.cursor(dictionary=True)
            cursor.execute(
                "SELECT model_id, model_name, category, specs, description, price, features "
                "FROM devices "
                "WHERE model_name LIKE %s OR category LIKE %s OR description LIKE %s "
                "LIMIT %s",
                (f"%{keyword}%", f"%{keyword}%", f"%{keyword}%", limit)
            )
            results = cursor.fetchall()
            cursor.close()
            conn.close()

            return [cls(**row) for row in results]
        except Exception as e:
            logger.error(f"[Device.search_by_keyword]搜索失败：{str(e)}")
            return []

    def to_dict(self) -> dict:
        """转换为字典"""
        return {
            "model_id": self.model_id,
            "model_name": self.model_name,
            "category": self.category,
            "specs": self.specs,
            "description": self.description,
            "price": self.price,
            "features": self.features
        }

    def __str__(self) -> str:
        """格式化输出"""
        import json
        specs = json.loads(self.specs) if self.specs else {}
        features = json.loads(self.features) if self.features else []

        result = f"【{self.model_name}】（型号：{self.model_id}）\n"
        result += f"- 类别：{self.category}\n"
        result += f"- 价格：￥{self.price}\n"
        if specs:
            result += f"- 规格："
            for key, value in list(specs.items())[:3]:  # 只显示前3个
                result += f"{key}:{value} "
            result += "\n"
        if features:
            result += f"- 特色功能：{', '.join(features[:3])}\n"
        return result

            return [cls(**row) for row in results]

        """检查库存可用数量"""
        try:
            conn = get_db_connection()
            cursor = conn.cursor(dictionary=True)

            if warehouse:
                cursor.execute(
                    "SELECT stock FROM inventory WHERE model_id = %s AND warehouse = %s",
                    (model_id, warehouse)

            else:
                cursor.execute(
                    "SELECT SUM(stock) as total FROM inventory WHERE model_id = %s",
                    (model_id,)
                )

            result = cursor.fetchone()
            cursor.close()
            conn.close()

            if warehouse:
                return result['stock'] if result else 0
            return result['total'] if result and result['total'] else 0
        except Exception as e:
            logger.error(f"[Inventory.check_available]查询失败：{str(e)}")
            return 0

    def to_dict(self) -> dict:
        """转换为字典"""

            "warehouse": self.warehouse,
```

```python
"""
数据库初始化脚本
执行SQL脚本，创建数据库和表
"""

import mysql.connector
from utils.path_tool import get_abs_path
from utils.logger_handler import logger
import yaml


def init_database():
    """初始化数据库"""
    # 加载配置
    config_path = get_abs_path("config/database.yml")
    with open(config_path, "r", encoding="utf-8") as f:
        config = yaml.load(f, Loader=yaml.FullLoader)
        mysql_config = config["mysql"]

    # 读取SQL脚本
    sql_path = get_abs_path("sql/init_schema.sql")
    with open(sql_path, "r", encoding="utf-8") as f:
        sql_script = f.read()

    # 连接MySQL（不指定数据库）
    try:
        conn = mysql.connector.connect(
            host=mysql_config["host"],
            port=mysql_config["port"],
            user=mysql_config["user"],
            password=mysql_config["password"],
            charset=mysql_config["charset"]
        )
        cursor = conn.cursor()

        # 执行SQL脚本
        for statement in sql_script.split(';'):
            statement = statement.strip()
            if statement:
                cursor.execute(statement)

        conn.commit()
        cursor.close()
        conn.close()

        logger.info("[init_database]数据库初始化成功！")
        print("✅ 数据库初始化成功！")

    except Exception as e:
        logger.error(f"[init_database]初始化失败：{str(e)}")
        print(f"❌ 数据库初始化失败：{str(e)}")
        raise e


if __name__ == "__main__":
    init_database()
```

```python
@tool(description="搜索设备，参数为keyword（关键词，如设备名称、类别），返回匹配的设备列表")

    搜索设备工具
    通过关键词（设备名称、类别）搜索相关设备
    """
```

```sql
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_session_created (session_id, created_at)
);
```

### 2.2 核心功能

1. **保存对话**：每次用户输入和AI响应后，写入数据库
2. **加载对话**：会话开始时，从数据库加载最近N条历史
3. **清空对话**：清空当前会话或用户的所有历史
4. **跨会话恢复**：用户登录后可恢复历史对话
